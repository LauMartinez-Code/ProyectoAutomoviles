create database ABM_Automoviles

use ABM_Automoviles

--Creacion de Tablas
create table Tipos_Clientes
(
	id_tipoCliente int identity(1,1),
	Tipo varchar(50) not null,
	constraint PK_TiposCientes primary key (id_tipoCliente)
)

create table Tipos_Documentos
(
	id_tipoDocumento int identity(1,1),
	Tipo varchar(40) not null,
	constraint PK_TiposDocumentos primary key (id_tipoDocumento)
)

create table Paises
(
	id_pais int identity(1,1),
	Pais varchar(50) not null,
	constraint PK_Paises primary key (id_pais)
)

create table Provincias
(
	id_provincia int identity(1,1),
	Provincia varchar(50) not null,
	id_pais int,
	constraint PK_Provincias primary key (id_provincia),
	constraint fk_Paises_Provincias foreign key (id_pais) references Paises(id_pais)
)

create table Localidades
(
	id_localidad int identity(1,1),
	Localidad varchar(60) not null,
	id_provincia int,
	constraint PK_Localidades primary key (id_localidad),
	constraint fk_Provincias_Localidades foreign key (id_provincia) references Provincias(id_provincia)
)

create table Barrios
(
	id_barrio int identity(1,1),
	Barrio varchar(60) not null,
	id_localidad int,
	constraint PK_Barrios primary key (id_barrio),
	constraint fk_Localidades_Barrios foreign key (id_localidad) references Localidades(id_localidad)
)

create table Tipos_Contactos
(
	id_tipoContacto int identity(1,1),
	Tipo varchar(40) not null,
	constraint PK_TiposContactos primary key (id_tipoContacto)
)

create table Clientes
(
	id_cliente int identity(1,1),
	Nombre varchar(80)not null,
	Apellido varchar(80),
	id_tipoDocumento int,
	nro_Documento bigint,
	id_tipoCliente int,
	Direccion varchar(110),
	id_barrio int,
	constraint PK_Clientes primary key (id_cliente),
	constraint fk_TiposDocumetnos_Clientes foreign key (id_tipoDocumento) references Tipos_Documentos(id_tipoDocumento),
	constraint fk_TiposClientes_Clientes foreign key (id_tipoCliente) references Tipos_Clientes(id_TipoCliente),
	constraint fk_Barrios_Clientes foreign key (id_barrio) references Barrios(id_barrio)
)

create table Contactos_Clientes
(
	id_contactoCliente int identity(1,1),
	id_cliente int,
	id_tipoContacto int,
	Contacto varchar(100) not null,
	constraint PK_ContactosClientes primary key (id_contactoCliente),
	constraint fk_Clientes_ContactosClientes foreign key (id_cliente) references Clientes(id_cliente),
	constraint fk_TiposDeContactos_ContactosClientes foreign key (id_tipoContacto) references Tipos_Contactos(id_tipoContacto)
)

create table Vendedores
(
	id_vendedor int identity(1,1),
	Nombre varchar(80)not null,
	Apellido varchar(80) not null,
	id_tipoDocumento int,
	nro_Documento bigint,
	Direccion varchar(110),
	id_barrio int,
	constraint PK_Vendedores primary key (id_vendedor),
	constraint fk_TiposDocumentos_Clientes foreign key (id_tipoDocumento) references Tipos_Documentos(id_tipoDocumento),
	constraint fk_Barrios_Vendedores foreign key (id_barrio) references Barrios(id_barrio)
)

create table Contactos_Vendedores
(
	id_contactoVendedor int identity(1,1),
	id_vendedor int,
	id_tipoContacto int,
	Contacto varchar(100) not null,
	constraint PK_ContactosVendedores primary key (id_contactoVendedor),
	constraint fk_Vendedores_ContactosVendedores foreign key (id_vendedor) references Vendedores(id_vendedor),
	constraint fk_TiposDeContactos_ContactosVendedores foreign key (id_tipoContacto) references Tipos_Contactos(id_tipoContacto)
)

create table Facturas
(
	id_factura int identity(1,1),
	Fecha datetime not null,
	Hora time not null,
	id_cliente int,
	id_vendedor int,
	constraint PK_Facturas primary key (id_factura),
	constraint fk_Vendedores_Facturas foreign key (id_vendedor) references Vendedores(id_vendedor),
	constraint fk_Cliente_Facturas foreign key (id_cliente) references Clientes(id_cliente),
)

create table Medios_De_Pago
(
	id_medio_de_pago int identity(1,1),
	Medio varchar(80) not null,
	constraint PK_MediosDePago primary key (id_medio_de_pago)
)

create table MediosDePago_Facturas
(
	id_pago int identity(1,1),
	id_factura int,
	id_medio_de_pago int,
	Monto smallmoney not null,
	constraint PK_MP_Facutras primary key (id_pago),
	constraint fk_MediosDePago_MedPagoFacturas foreign key (id_medio_de_pago) references Medios_De_Pago(id_medio_de_pago),
	constraint fk_Facturas_MedPagoFacturas foreign key (id_factura) references Facturas(id_factura)
)

create table Marcas
(
	id_marca int identity(1,1),
	Marca varchar(30) not null,
	constraint PK_Marcas primary key (id_marca)
)


create table Colores
(
	id_Color int identity(1,1),
	Color varchar(50) not null,
	Cod_color_Hexa varchar(8),
	constraint PK_Colores primary key (id_color)
)

create table Tipos_Cajas_Velocidad
(
	id_cajaVelocidad int identity(1,1),
	Tipo varchar(20) not null,
	constraint PK_TiposCajasVelocidad primary key (id_cajaVelocidad)
)

create table Tipos_Direcciones
(
	id_tipoDireccion int identity(1,1),
	Tipo varchar(40) not null,
	constraint PK_TiposDirecciones primary key(id_TipoDireccion)
)

create table Combustibles
(
	id_combustible int identity(1,1),
	Combustible varchar(20) not null,
	constraint PK_Combustibles primary key(id_combustible)
)

create table Autos
(
	id_automovil int identity(1,1),
	id_marca int,
	Modelo varchar(30) not null,
	Año int,
	Dimenciones varchar(30),
	Motor varchar(20),
	id_combustible int,
	id_cajaVelocidad int,
	Velocidades tinyint,
	id_tipoDireccion int,
	id_Color int,
	Cantidad_puertas tinyint,
	Tamaño_llantas char(5),
	Cantidad_AirBags tinyint,
	Camara_de_Retroceso bit,
	Sensores_lluvia_parabrisas bit,
	Techo_cielo bit,
	Climatizador_biZona bit,
	LevantaVidrio_automatico bit,
	Audio_Conectividad varchar(100),
	Cierre_Centralizado bit,
	constraint PK_Autos primary key (id_automovil),
	constraint fk_Marcas_Autos foreign key (id_marca) references Marcas(id_marca),
	constraint fk_Combustibles_Autos foreign key (id_combustible) references Combustibles(id_combustible),
	constraint fk_TiposCajasVelocidad_Autos foreign key (id_cajaVelocidad) references Tipos_Cajas_Velocidad(id_cajaVelocidad),
	constraint fk_TiposDirecciones_Autos foreign key (id_TipoDireccion) references Tipos_Direcciones(id_tipoDireccion),
	constraint fk_Colores_Autos foreign key (id_Color) references Colores(id_Color)
)

create table Autopartes
(
	id_autoparte int identity(1,1),
	id_marca int,
	Modelo varchar(60) not null,
	id_pais_Origen int,
	descripcion varchar(90),
	Stock int,
	constraint PK_Autopartes primary key (id_autoparte),
	constraint fk_Marcas_Autopartes foreign key (id_marca) references Marcas(id_marca),
	constraint fk_Paises_Autopartes foreign key (id_pais_Origen) references Paises(id_pais)
)

create table Tipos_Productos
(
	id_tipoProducto int identity(1,1),
	Tipo varchar(30),
	constraint PK_TiposProductos primary key (id_tipoProducto)
)


create table Productos
(
	id_producto int identity (1,1),
	id_tipoProducto int,
	id_automovil int,
	id_autoparte int,
	Precio money,
	constraint PK_id_producto primary key (id_producto),
	constraint fk_Autos_Productos foreign key (id_automovil) references Autos(id_automovil),
	constraint fk_Autopartes_Productos foreign key (id_autoparte) references Autopartes(id_autoparte),
	constraint fk_TiposProductos_Productos foreign key (id_tipoProducto) references Tipos_Productos(id_tipoProducto),
)

create table Detalle_Factura
(
	id_detalle_factura int identity(1,1),
	id_factura int,
	id_producto int,
	Cantidad int,
	Precio money,
	constraint PK_DetalleFactura primary key (id_detalle_factura),
	constraint fk_Facturas_DetalleFactura foreign key (id_factura) references Facturas(id_factura),
	constraint fk_Productos_DetalleFactura foreign key (id_producto) references Productos(id_producto)
)

create table Ordenes_de_Pedidos
(
	id_orden_pedido int identity(1,1),
	id_factura int,
	fecha_entrega date,
	forma_de_entrega varchar(100),
	constraint PK_OrdenPedido primary key (id_orden_pedido),
	constraint fk_Facturas_OrdenesDePedidos foreign key (id_factura) references Facturas(id_factura)
)

create table Detalle_Pedido
(
	id_detalle_pedido int identity(1,1),
	id_orden_pedido int,
	id_producto int,
	Cantidad int,
	Descripcion varchar(60),
	constraint PK_id_detalle_pedido primary key (id_detalle_pedido),
	constraint fk_OrdenPedido_DetallePedido foreign key (id_orden_pedido) references Ordenes_de_Pedidos(id_orden_pedido),
	constraint fk_Productos_DetallePedido foreign key (id_producto) references Productos(id_producto)
)

create table Historial
(
	id_historial int identity(1,1),
	id_producto int,
	Fecha date,
	constraint PK_Historial primary key (id_historial),
	constraint fk_Productos_Historial foreign key (id_producto) references Productos(id_producto)
)

create view viewCargarAutos
as
select a.id_automovil, a.id_marca, a.Modelo, a.Año, a.Dimenciones, a.Motor,a.id_combustible,
	a.id_cajaVelocidad, a.Velocidades, a.id_tipoDireccion, a.id_Color, a.Cantidad_puertas, a.Tamaño_llantas,
	a.Cantidad_AirBags, a.Camara_de_Retroceso, a.Sensores_lluvia_parabrisas, a.Techo_cielo, a.Climatizador_biZona,
	a.LevantaVidrio_automatico, a.Audio_Conectividad, a.Cierre_Centralizado, p.id_producto
from Productos p right join Autos a on p.id_automovil = a.id_automovil

go

create view viewMostrarAutos
as
select a.id_automovil Codigo, m.Marca, a.Modelo, a.Año, a.Dimenciones, a.Motor,
	c.Combustible, tcv.Tipo 'Caja Velocidad', a.Velocidades, td.Tipo 'Tipo de Direccion',
	col.Color, a.Cantidad_puertas 'Cant. Puertas', a.Tamaño_llantas 'Tamaño Llantas',
	a.Cantidad_AirBags 'Cant. AirBags', a.Camara_de_Retroceso 'Camara de Retroceso',
	a.Sensores_lluvia_parabrisas 'Sensores de Lluvia', a.Techo_cielo 'Techo Cielo',
	a.Climatizador_biZona 'Climatizador BiZona', a.LevantaVidrio_automatico 'Levanta Vidrio Autom.',
	a.Cierre_Centralizado 'Cierre Centralizado', a.Audio_Conectividad 'Audio y Conectividad'
from Autos a join marcas m on m.id_marca = a.id_marca
	join Combustibles c on c.id_combustible = a.id_combustible
	join Tipos_Cajas_Velocidad tcv on tcv.id_cajaVelocidad = a.id_cajaVelocidad
	join Tipos_Direcciones td on td.id_tipoDireccion = a.id_tipoDireccion
	join Colores col on col.id_Color = a.id_Color

create view viewCargarAutos
as
select a.id_automovil, a.id_marca, a.Modelo, a.Año, a.Dimenciones, a.Motor,a.id_combustible,
	a.id_cajaVelocidad, a.Velocidades, a.id_tipoDireccion, a.id_Color, a.Cantidad_puertas, a.Tamaño_llantas,
	a.Cantidad_AirBags, a.Camara_de_Retroceso, a.Sensores_lluvia_parabrisas, a.Techo_cielo, a.Climatizador_biZona,
	a.LevantaVidrio_automatico, a.Audio_Conectividad, a.Cierre_Centralizado, p.id_producto
from Productos p right join Autos a on p.id_automovil = a.id_automovil

go

create view viewMostrarAutos
as
select a.id_automovil Codigo, m.Marca, a.Modelo, a.Año, a.Dimenciones, a.Motor,
	c.Combustible, tcv.Tipo 'Caja Velocidad', a.Velocidades, td.Tipo 'Tipo de Direccion',
	col.Color, a.Cantidad_puertas 'Cant. Puertas', a.Tamaño_llantas 'Tamaño Llantas',
	a.Cantidad_AirBags 'Cant. AirBags', a.Camara_de_Retroceso 'Camara de Retroceso',
	a.Sensores_lluvia_parabrisas 'Sensores de Lluvia', a.Techo_cielo 'Techo Cielo',
	a.Climatizador_biZona 'Climatizador BiZona', a.LevantaVidrio_automatico 'Levanta Vidrio Autom.',
	a.Cierre_Centralizado 'Cierre Centralizado', a.Audio_Conectividad 'Audio y Conectividad'
from Autos a join marcas m on m.id_marca = a.id_marca
	join Combustibles c on c.id_combustible = a.id_combustible
	join Tipos_Cajas_Velocidad tcv on tcv.id_cajaVelocidad = a.id_cajaVelocidad
	join Tipos_Direcciones td on td.id_tipoDireccion = a.id_tipoDireccion
	join Colores col on col.id_Color = a.id_Color
